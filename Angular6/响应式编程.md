    @NgModule({
    imports: [
        BrowserModule,
        FormsModule,
        ReactiveFormsModule
    })

    <input [formControl]="titleField" >

    import {FormControl} from '@angular/forms';
    import 'rxjs/Rx';

    class xxComponent {
        keyword;

        titleField: FormControl = new FormControl();
        constructor() {
            this.titleField.valueChanges
                .debounceTime(500)
                .subscribe( value => {
                    this.keyword = value;
                } );
        }
    }
